def maxProfit(prices):
    buy1 = buy2 = float('-inf')
    sell1 = sell2 = 0
    for p in prices:
        buy1  = max(buy1,  -p)
        sell1 = max(sell1, buy1 + p)
        buy2  = max(buy2,  sell1 - p)
        sell2 = max(sell2, buy2 + p)
    return sell2
# Time: O(n), Space: O(1)

def lengthOfLongestSubstring(s):
    char_idx = {}
    left = max_len = 0
    for right, ch in enumerate(s):
        if ch in char_idx and char_idx[ch] >= left:
            left = char_idx[ch] + 1
        char_idx[ch] = right
        max_len = max(max_len, right - left + 1)
    return max_len
# Time: O(n), Space: O(min(m,n))

Implement LRU Cache with O(1) get and put operations. (Very commonly reported at JPMC)
DesignHashMap+DLLJPMC Classic
▼ Show
Model Answer
Use a doubly linked list (O(1) removal) + HashMap (O(1) lookup).

from collections import OrderedDict

class LRUCache:
    def __init__(self, capacity):
        self.cap = capacity
        self.cache = OrderedDict()

    def get(self, key):
        if key not in self.cache: return -1
        self.cache.move_to_end(key)
        return self.cache[key]

    def put(self, key, value):
        if key in self.cache:
            self.cache.move_to_end(key)
        self.cache[key] = value
        if len(self.cache) > self.cap:
            self.cache.popitem(last=False)
# Both get/put: O(1) amortized
Tip: Know the manual DLL implementation too — interviewers may ask you to implement without OrderedDict.

import time
from collections import defaultdict, deque

class RateLimiter:
    def __init__(self, max_requests, window_sec):
        self.max_req = max_requests
        self.window  = window_sec
        self.users   = defaultdict(deque)

    def is_allowed(self, user_id):
        now = time.time()
        q = self.users[user_id]
        while q and now - q[0] >= self.window:
            q.popleft()
        if len(q) < self.max_req:

            return True
        return False

▼ Show
Model Answer

import heapq
def findKthLargest(nums, k):
    return heapq.nlargest(k, nums)[-1]
QuickSelect O(n) average:
def findKthLargest(nums, k):
    pivot = nums[len(nums)//2]
    left  = [x for x in nums if x > pivot]
    mid   = [x for x in nums if x == pivot]
    right = [x for x in nums if x < pivot]
    if k <= len(left): return findKthLargest(left, k)
    if k <= len(left)+len(mid): return pivot
    return findKthLargest(right, k-len(left)-len(mid))
Trade-offs: Heap = O(n log k) time, O(k) space — good for streaming. QuickSelect = O(n) avg, O(n²) worst — good for in-memory.

def numIslands(grid):
    if not grid: return 0
    rows, cols = len(grid), len(grid[0])
    count = 0
    def dfs(r, c):
        if r < 0 or r >= rows or c < 0 or c >= cols or grid[r][c] != '1': return
        grid[r][c] = '0'
        for dr, dc in [(0,1),(0,-1),(1,0),(-1,0)]: dfs(r+dr, c+dc)
    for r in range(rows):
        for c in range(cols):
            if grid[r][c] == '1': dfs(r, c); count += 1
    return count

Find the median of a data stream in real-time. (Reported in JPMC senior interviews)
HeapHardJPMC Senior
▼ Show
Model Answer
Two heaps: max-heap for lower half, min-heap for upper half.

import heapq

class MedianFinder:
    def __init__(self):
        self.lo = []  # max-heap (negate)
        self.hi = []  # min-heap

    def addNum(self, num):
        heapq.heappush(self.lo, -num)
        heapq.heappush(self.hi, -heapq.heappop(self.lo))
        if len(self.hi) > len(self.lo):
            heapq.heappush(self.lo, -heapq.heappop(self.hi))

    def findMedian(self):
        if len(self.lo) > len(self.hi): return -self.lo[0]
        return (-self.lo[0] + self.hi[0]) / 2
# addNum: O(log n), findMedian: O(1)

import heapq

def minMeetingRooms(intervals):
    if not intervals: return 0
    intervals.sort(key=lambda x: x[0])
    heap = []
    for start, end in intervals:
        if heap and heap[0] <= start:
            heapq.heapreplace(heap, end)
        else:
            heapq.heappush(heap, end)
    return len(heap)
# Time: O(n log n), Space: O(n)
Intuition: Sort by start. Min-heap of end times. If earliest-ending meeting ended before new one starts, reuse that room.

Design a data structure supporting insert, delete, and getRandom in O(1) time.
DesignHashMap+ArrayJPMC Reported
▼ Show
Model Answer
import random

class RandomizedSet:
    def __init__(self):
        self.val_to_idx = {}
        self.vals = []

    def insert(self, val):
        if val in self.val_to_idx: return False
        self.vals.append(val)
        self.val_to_idx[val] = len(self.vals) - 1
        return True

    def remove(self, val):
        if val not in self.val_to_idx: return False
        idx = self.val_to_idx[val]
        last = self.vals[-1]
        self.vals[idx] = last
        self.val_to_idx[last] = idx
        self.vals.pop()
        del self.val_to_idx[val]
        return True

    def getRandom(self):
        return random.choice(self.vals)
Key insight: Swap element to delete with last element, then pop — O(1) delete without gaps.

Implement a Trie. Add search and startsWith. When is Trie better than HashMap?

class TrieNode:
    def __init__(self): self.children = {}; self.is_end = False

class Trie:
    def __init__(self): self.root = TrieNode()

    def insert(self, word):
        node = self.root
        for ch in word: node = node.children.setdefault(ch, TrieNode())
        node.is_end = True

    def search(self, word):
        node = self.root
        for ch in word:
            if ch not in node.children: return False
            node = node.children[ch]
        return node.is_end

    def startsWith(self, prefix):
        node = self.root
        for ch in prefix:
            if ch not in node.children: return False
            node = node.children[ch]
        return True
Trie vs HashMap: Trie is better for prefix queries, autocomplete, and enumerating all words with a given prefix.

def minDistance(word1, word2):
    m, n = len(word1), len(word2)
    dp = list(range(n+1))
    for i in range(1, m+1):
        prev = dp[0]; dp[0] = i
        for j in range(1, n+1):
            temp = dp[j]
            if word1[i-1] == word2[j-1]: dp[j] = prev
            else: dp[j] = 1 + min(prev, dp[j], dp[j-1])
            prev = temp
    return dp[n]
# Time: O(mn), Space: O(n)

Shortest path in a weighted graph (Dijkstra). When would you use Bellman-Ford instead?
GraphsDijkstraJPMC Reported
▼ Show
Model Answer
import heapq

def dijkstra(graph, src):
    dist = {node: float('inf') for node in graph}
    dist[src] = 0; pq = [(0, src)]
    while pq:
        d, u = heapq.heappop(pq)
        if d > dist[u]: continue
        for v, w in graph[u]:
            if dist[u] + w < dist[v]:
                dist[v] = dist[u] + w
                heapq.heappush(pq, (dist[v], v))
    return dist
Dijkstra vs Bellman-Ford: Dijkstra O((V+E)logV), non-negative weights only. Bellman-Ford O(VE), handles negative weights, detects negative cycles.
Finance: Bellman-Ford for currency arbitrage detection (negative cycle = arbitrage opportunity).

from collections import Counter

def findAnagrams(s, p):
    result = []; p_count = Counter(p); window = Counter(s[:len(p)])
    if window == p_count: result.append(0)
    for i in range(len(p), len(s)):

def searchRange(nums, target):
    def find_bound(is_left):
        lo, hi, bound = 0, len(nums)-1, -1
        while lo <= hi:
            mid = (lo + hi) // 2
            if nums[mid] == target:
                bound = mid
                if is_left: hi = mid - 1
                else: lo = mid + 1
            elif nums[mid] < target: lo = mid + 1
            else: hi = mid - 1
        return bound
    return [find_bound(True), find_bound(False)]

def wordBreak(s, wordDict):
    word_set = set(wordDict)
    n = len(s)
    dp = [False] * (n + 1)
    dp[0] = True
    for i in range(1, n + 1):
        for j in range(i):
            if dp[j] and s[j:i] in word_set:
                dp[i] = True; break
    return dp[n]
# Time: O(n^2), Space: O(n)

What is Python's memory model? Explain mutable vs immutable objects and how assignment works under the hood.
Python InternalsMemory
▼ Show
Model Answer
In Python, everything is an object. Variables are references (pointers) to objects, not containers.

Immutable: int, float, str, tuple — cannot be changed after creation. Reassignment creates a new object.
Mutable: list, dict, set — can be modified in place. Multiple variables can reference the same object.

# Mutable — shared reference!
a = [1,2,3]; b = a; b.append(4)
# a is now [1,2,3,4] — same object!

# Python interns small ints (-5 to 256)
a = 256; b = 256; print(a is b)  # True
a = 257; b = 257; print(a is b)  # False

What are Python decorators? Write one that logs execution time and handles exceptions.
PythonDecorators
▼ Show
Model Answer
A decorator is a higher-order function that wraps another function to extend its behavior without modifying it.

import time, logging, functools

def timed_logger(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        start = time.perf_counter()
        try:
            result = func(*args, **kwargs)
            logging.info(f"{func.__name__} took {time.perf_counter()-start:.3f}s")
            return result
        except Exception as e:
            logging.error(f"{func.__name__} failed: {e}")
            raise
    return wrapper
Key: Always use @functools.wraps(func) to preserve the original function's metadata.

Explain Python's GIL. How does it affect multi-threaded applications? How do you handle CPU-bound parallelism?
ConcurrencyGIL
▼ Show
Model Answer
The GIL (Global Interpreter Lock) is a mutex in CPython that allows only one thread to execute Python bytecode at a time, even on multi-core systems.

Impact:
  • I/O-bound tasks: Threading still helps — GIL is released during I/O waits (network, file, DB)
  • CPU-bound tasks: Threading provides NO speedup — use multiprocessing instead
Solutions for CPU-bound parallelism:
  • multiprocessing — separate processes, each with own GIL
  • concurrent.futures.ProcessPoolExecutor — high-level API
  • NumPy/SciPy — release GIL for C-level operations
  • Cython with nogil blocks
  • PySpark — distributed processing bypasses GIL entirely
JPMC Context: Risk calculations (Monte Carlo simulations) are CPU-bound → multiprocessing. Market data ingestion is I/O-bound → asyncio or threading.

asyncio uses a single-threaded event loop with cooperative multitasking — coroutines voluntarily yield control at await points. Threading uses OS-level threads with preemptive switching.

        tasks = [fetch_price(session, t) for t in tickers]

Use asyncio when: High-concurrency I/O (APIs, WebSockets, DB queries). Use threading for legacy blocking code.

def memoize(func):
    cache = {}
    def wrapper(*args):
        if args not in cache:
            cache[args] = func(*args)
        return cache[args]
    return wrapper

@memoize
def fib(n):
    if n < 2: return n
    return fib(n-1) + fib(n-2)

Explain ACID properties and database isolation levels. What are dirty read, non-repeatable read, and phantom read?
DatabaseACID
▼ Show
Model Answer
ACID: Atomicity, Consistency, Isolation, Durability.

Isolation levels (weakest to strongest):
READ UNCOMMITTED: Dirty read — T1 reads uncommitted data from T2. T2 rolls back. T1 saw data that never existed.
READ COMMITTED: Prevents dirty reads. Non-repeatable read — T1 reads balance=$1000. T2 commits debit. T1 reads again=$500.
REPEATABLE READ: Prevents non-repeatable reads. Phantom read — T1 counts 10 rows. T2 inserts one. T1 counts 11.
SERIALIZABLE: Prevents all anomalies. Highest isolation, lowest concurrency.

JPMC recommendation: READ COMMITTED for most operations. SERIALIZABLE for critical financial operations (balance transfers).

What is Value at Risk (VaR)? How would you implement a VaR calculation system in Python?
Risk ManagementQuantitative
▼ Show
Model Answer
VaR: Maximum loss expected over a given time period at a given confidence level. E.g., "1-day 99% VaR of $1M means 1% chance of losing more than $1M in a day."

Three methods: Historical simulation (no distributional assumptions), Parametric (assumes normal distribution, fast), Monte Carlo (most flexible, expensive).

import numpy as np
from scipy.stats import norm

def historical_var(returns, confidence=0.99):
    return np.percentile(returns, (1 - confidence) * 100)

def parametric_var(portfolio_value, volatility, confidence=0.99, days=1):
    z_score = norm.ppf(1 - confidence)
    return portfolio_value * volatility * z_score * np.sqrt(days)
Limitation: VaR doesn't tell you how bad losses can be beyond the threshold. Use CVaR (Expected Shortfall) for tail risk.

from abc import ABC, abstractmethod

class Observer(ABC):
    @abstractmethod
    def update(self, event): ...

class PriceSubject:
    def __init__(self): self._observers = []

    def subscribe(self, obs): self._observers.append(obs)
    def unsubscribe(self, obs): self._observers.remove(obs)

    def set_price(self, ticker, price):
        for obs in self._observers:
            obs.update({'ticker': ticker, 'price': price})

class AlertSystem(Observer):

        if event['price'] > 1000:
            print(f"ALERT: {event['ticker']} hit ${event['price']}")

What is the Factory pattern? Implement a financial instrument factory.
Design PatternsFinance
▼ Show
Model Answer
class InstrumentFactory:
    _registry = {}

    @classmethod
    def register(cls, instrument_type):
        def decorator(klass):
            cls._registry[instrument_type] = klass
            return klass
        return decorator

    @classmethod
    def create(cls, instrument_type, **kwargs):
        if instrument_type not in cls._registry:
            raise ValueError(f"Unknown: {instrument_type}")
        return cls._registry[instrument_type](**kwargs)

@InstrumentFactory.register('equity')
class Equity:
    def __init__(self, ticker, exchange): ...

@InstrumentFactory.register('bond')
class Bond:
    def __init__(self, isin, coupon, maturity): ...

equity = InstrumentFactory.create('equity', ticker='AAPL', exchange='NASDAQ')
Advantage: Adding new instrument types requires only registering a new class — Open/Closed Principle.

What is the difference between composition and inheritance? When should you prefer each?

Model Answer
Inheritance: "is-a" relationship. Subclass inherits behavior from parent. Creates tight coupling.
Composition: "has-a" relationship. Class contains instances of other classes. Loose coupling — components can be swapped independently.

Prefer composition when: Behavior needs to vary at runtime, multiple inheritance would be needed, the "is-a" relationship isn't truly accurate.

# Inheritance (fragile — class explosion)
class PremiumSavingsAccount(SavingsAccount): ...

class Account:
    def __init__(self, interest_calc, overdraft_policy):
        self.interest = interest_calc
        self.overdraft = overdraft_policy
Rule: "Favor composition over inheritance" (Gang of Four). Use inheritance only for true "is-a" relationships.

Implement the Command pattern for an undo/redo system for financial transactions.

from abc import ABC, abstractmethod

class Command(ABC):
    @abstractmethod
    def execute(self): ...
    @abstractmethod
    def undo(self): ...

class TransferCommand(Command):
    def __init__(self, from_acc, to_acc, amount):
        self.from_acc = from_acc; self.to_acc = to_acc; self.amount = amount

    def execute(self):
        self.from_acc.debit(self.amount); self.to_acc.credit(self.amount)

    def undo(self):
        self.to_acc.debit(self.amount); self.from_acc.credit(self.amount)

class TransactionHistory:
    def __init__(self): self.history = []
    def execute(self, cmd): cmd.execute(); self.history.append(cmd)
    def undo_last(self):
        if self.history: self.history.pop().undo()
